/**
 * @namespace {object} Util
 * @description
 * Contains a set of general utility methods
 */
module.exports = {
	/**
	 * @function
	 * @name Util.proxy
	 * @param {Function} fn the function to proxy
	 * @param {object} context the context on which the function should be called
	 * @description 
	 * This is a method
	 * 
	 */
  proxy:function(fn,ctx){
  	var c = this || ctx;
  	return function(){
  		return fn.apply(c,arguments);
  	}
  }
}
